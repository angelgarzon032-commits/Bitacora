Bot en HTML + JavaScript que lista los conceptos encontrados en un código

Te paso un archivo HTML (single-file) que actúa como un "bot" en la página: pegas tu código, eliges el lenguaje (o "Auto"), marcas la opción "No alterar el código" si no quieres que el bot cambie nada, y pulsa Analizar. El script no modifica tu código: sólo lo lee y extrae conceptos (funciones, clases, imports, variables, comentarios, estructuras de control, TODOs, tipos comunes, llamadas a librerías, constantes, etc.) y te muestra una lista ordenada con ejemplos y conteos. También puedes descargar la lista como CSV.

Copia todo esto en un archivo analizador_conceptos.html y ábrelo en tu navegador.

    // ---------- Análisis principal ----------
    function analyzeCode(code, langHint='auto') {
      const res = {
        meta: { length: code.length, lines: code.split('\\n').length },
        detectedLang: langHint,
        concepts: {}
      };

      const addConcept = (name, example) => {
        if (!res.concepts[name]) res.concepts[name] = { count:0, examples: new Set() };
        res.concepts[name].count++;
        if (example) {
          // store unique examples, limit to 6
          if (res.concepts[name].examples.size < 6) res.concepts[name].examples.add(example.trim());
        }
      };

      // language detection (lightweight)
      const lowered = code.slice(0,2000).toLowerCase();
      let detected = langHint === 'auto' ? null : langHint;
      if (!detected) {
        if (/\\bdef\\b|:\\n\\s+/.test(code)) detected = 'python';
        else if (/\\bimport\\s.+from\\s+|require\\(|function\\s+|=>|console\\.log/.test(code)) detected = 'javascript';
        else if (/#include\\s+<|printf\\(|scanf\\(|int\\s+main\\(/.test(code)) detected = 'c';
        else if (/package\\s+|public\\s+class\\s+|System\\.out\\.println/.test(code)) detected = 'java';
        else if (/<\\/?html|<script|<div|<!doctype html/i.test(code)) detected = 'html';
        else detected = 'unknown';
      }
      res.detectedLang = detected;

      // Common patterns
      const common = patterns.common;

      // TODO/FIXME
      let m;
      while ((m = common.todo.exec(code)) !== null) {
        addConcept('TODO/FIXME', m[0]);
      }

      // Comments (count lines starting with comment markers)
      const commentMatches = code.match(common.commentLine);
      if (commentMatches) {
        addConcept('Comentarios', commentMatches.slice(0,6).join('\\n'));
      }

      // Imports/includes
      while ((m = common.include.exec(code)) !== null) {
        addConcept('Imports / Includes', m[0].trim());
      }

      // Functions (varios patrones)
      const fnPatterns = [...common.functionGeneric, common.defPython];
      fnPatterns.forEach(pat => {
        pat.lastIndex = 0;
        while ((m = pat.exec(code)) !== null) {
          const name = m[1] || m[0];
          addConcept('Funciones / Métodos', name);
        }
      });

      // Classes
      common.class.lastIndex = 0;
      while ((m = common.class.exec(code)) !== null) {
        addConcept('Clases', m[1]);
      }

      // Variables JS
      common.varJs.lastIndex = 0;
      while ((m = common.varJs.exec(code)) !== null) {
        addConcept('Variables (var/let/const)', m[2]);
      }

      // Variables estilo C-like (type name ;)
      common.varCLike.lastIndex = 0;
      while ((m = common.varCLike.exec(code)) !== null) {
        // avoid catching for loops (int i = 0) too often, still it's useful
        addConcept('Declaraciones con tipo (C/Java/Go/C#)', m[0].trim());
      }

      // Loops, if/else, try/catch, return
      ['loop','ifelse','trycatch','returnStmt','asyncAwait','lambda'].forEach(key => {
        const p = common[key];
        p.lastIndex = 0;
        let count = 0;
        while (p.exec(code)) count++;
        if (count) addConcept({
          loop: 'Bucles (for/while/foreach/do)',
          ifelse: 'Condicionales (if/else/switch)',
          trycatch: 'Manejo de excepciones (try/catch/throw)',
          returnStmt: 'Sentencias return',
          asyncAwait: 'Async / Await',
          lambda: 'Lambdas / expresiones inline'
        }[key], `${count} ocurrencias`);
      });

      // Regex literals
      const regexes = code.match(common.regexLiteral);
      if (regexes) addConcept('Expresiones regulares (literales)', regexes.slice(0,4).join(', '));

      // Language-specific
      const langSpec = patterns.languageSpecific[detected] || {};
      Object.keys(langSpec).forEach(k => {
        const pat = langSpec[k];
        pat.lastIndex = 0;
        let c;
        while ((c = pat.exec(code)) !== null) {
          addConcept((k[0].toUpperCase() + k.slice(1)).replace(/([A-Z])/g, ' $1').trim(), c[0].trim());
        }
      });

      // HTML tags
      if (detected === 'html') {
        const tagPat = patterns.languageSpecific.html.tag;
        tagPat.lastIndex = 0;
        let tags = new Set();
        while ((m = tagPat.exec(code)) !== null) tags.add(m[1]);
        if (tags.size) addConcept('Etiquetas HTML', Array.from(tags).slice(0,10).join(', '));
      }

      // heurística extra: detectar llamadas comunes a librerías
      const libCalls = [];
      const libRegex = /([A-Za-z_\\$][A-Za-z0-9_\\$]*)\\.(\\w+)\\s*\\(/g;
      while ((m = libRegex.exec(code)) !== null) {
        libCalls.push(m[1] + '.' + m[2] + '()');
      }
      if (libCalls.length) addConcept('Llamadas a métodos (obj.metodo())', libCalls.slice(0,6).join(', '));

      // constantes en mayúsculas
      const consts = code.match(/\\b[A-Z_]{2,}\\b/g);
      if (consts) addConcept('Constantes (MAYÚSCULAS)', consts.slice(0,6).join(', '));

      // Extra: detectar patrones de SQL embebido (SELECT, INSERT)
      const sql = code.match(/\\b(SELECT|INSERT|UPDATE|DELETE)\\b[\\s\\S]{1,80}/gi);
      if (sql) addConcept('SQL embebido', sql.slice(0,4).join(' | '));

      // Convert Set of examples to Array
      Object.keys(res.concepts).forEach(k => {
        res.concepts[k].examples = Array.from(res.concepts[k].examples);
      });

      return res;
    }

Breve explicación y recomendaciones:

  • El analizador es heurístico: usa expresiones regulares para encontrar patrones comunes. Para resultados más precisos habría que integrar un parser real por lenguaje (es posible, pero requiere más código y/o servicios).

  • La opción "No alterar código" está marcada por defecto: el bot nunca modifica tu código en la caja — sólo lo lee.

  • Si quieres que detecte más conceptos (p. ej. dependencias en package.json, anotaciones específicas, frameworks, patrones de diseño), dime qué buscas y te añado reglas específicas.

  • Si quieres que guarde los resultados en un archivo distinto (JSON, Excel, Word) te lo puedo añadir.

      const add = (name, example) => {
        if (!res.concepts[name]) res.concepts[name] = { count:0, examples: new Set() };
        res.concepts[name].count++;
        if (example && res.concepts[name].examples.size < 8) res.concepts[name].examples.add(example.trim());
      };

      // detección ligera de lenguaje

        else detected = 'unknown';

      res.detectedLang = detected;

      const common = patterns.common;
      let m;

      // TODOs
      while ((m = common.todo.exec(code)) !== null) add('TODO / FIXME', m[0]);

      // Comentarios
      const commentMatches = code.match(common.commentLine);
      if (commentMatches) add('Comentarios', commentMatches.slice(0,6).join('\\n'));

      // Imports / includes
      while ((m = common.include.exec(code)) !== null) add('Imports / Includes', m[0].trim());

      // Funciones / métodos (varios patrones)
      const fnPatterns = [...common.functionGeneric, common.defPython];
      fnPatterns.forEach(pat => {
        pat.lastIndex = 0;
        while ((m = pat.exec(code)) !== null) {
          add('Funciones / Métodos', m[1] || m[0]);
        }
      });

      // Clases
      common.class.lastIndex = 0;
      while ((m = common.class.exec(code)) !== null) add('Clases', m[1]);

      // Variables JS
      common.varJs.lastIndex = 0;
      while ((m = common.varJs.exec(code)) !== null) add('Variables (var/let/const)', m[2]);

      // Declaraciones con tipo (C/Java/Go/C#)
      common.varCLike.lastIndex = 0;
      while ((m = common.varCLike.exec(code)) !== null) add('Declaraciones con tipo (C/Java/Go/C#)', m[0].trim());

          asyncAwait:'Async / Await',

      });

      // Regex literals
      const regexes = code.match(common.regexLiteral);
      if (regexes) add('Expresiones regulares (literales)', regexes.slice(0,4).join(', '));

      // Language-specific
      const langSpec = patterns.languageSpecific[detected] || {};
      Object.keys(langSpec).forEach(k => {
        const pat = langSpec[k]; pat.lastIndex = 0;
        let c;
        while ((c = pat.exec(code)) !== null) add((k[0].toUpperCase()+k.slice(1)).replace(/([A-Z])/g,' $1').trim(), c[0].trim());
      });

      // HTML tags
      if (detected === 'html') {
        const tagPat = patterns.languageSpecific.html.tag; tagPat.lastIndex = 0;
        const tags = new Set();
        while ((m = tagPat.exec(code)) !== null) tags.add(m[1]);
        if (tags.size) add('Etiquetas HTML', Array.from(tags).slice(0,20).join(', '));
      }

      const libRegex = /([A-Za-z_\\$][A-Za-z0-9_\\$]*)\\.(\\w+)\\s*\\(/g;
